<script>
function createTables(){
  const p = parseInt(document.getElementById("proc").value);
  const r = parseInt(document.getElementById("res").value);
  let html = "<h3>Allocation</h3>"+createMatrix("alloc",p,r);
  html += "<h3>Maximum</h3>"+createMatrix("max",p,r);
  html += "<h3>Available</h3>"+createRow("avail",r);
  document.getElementById("tables").innerHTML = html;
}

function createMatrix(prefix,p,r){
  let t="<table>";
  for(let i=0;i<p;i++){
    t+="<tr>";
    for(let j=0;j<r;j++)
      t+=`<td><input type='number' value='0' id='${prefix}-${i}-${j}'></td>`;
    t+="</tr>";
  }
  return t+"</table>";
}

function createRow(prefix,r){
  let t="<table><tr>";
  for(let j=0;j<r;j++)
    t+=`<td><input type='number' value='0' id='${prefix}-${j}'></td>`;
  return t+"</tr></table>";
}

function checkSafety(){
  const p = parseInt(document.getElementById("proc").value);
  const r = parseInt(document.getElementById("res").value);

  let alloc=[], max=[], need=[], avail=[], finish=[];
  for(let i=0;i<p;i++){
    alloc[i]=[]; max[i]=[]; need[i]=[]; finish[i]=false;
    for(let j=0;j<r;j++){
      alloc[i][j]=+document.getElementById(`alloc-${i}-${j}`).value;
      max[i][j]=+document.getElementById(`max-${i}-${j}`).value;
      need[i][j]=max[i][j]-alloc[i][j];
    }
  }
  for(let j=0;j<r;j++)
    avail[j]=+document.getElementById(`avail-${j}`).value;

  let work=[...avail], seq=[], count=0;
  while(count<p){
    let found=false;
    for(let i=0;i<p;i++){
      if(!finish[i]){
        let ok=true;
        for(let j=0;j<r;j++)
          if(need[i][j]>work[j]) ok=false;
        if(ok){
          for(let j=0;j<r;j++) work[j]+=alloc[i][j];
          finish[i]=true;
          seq.push("P"+i);
          count++; found=true;
        }
      }
    }
    if(!found) break;
  }

  const out=document.getElementById("output");
  if(seq.length===p){
    out.style.borderLeftColor="var(--success)";
    out.textContent="✅ SAFE STATE\nSafe Sequence: "+seq.join(" → ");
  }else{
    out.style.borderLeftColor="var(--danger)";
    out.textContent="❌ UNSAFE STATE\nAllocation prevented to avoid deadlock";
  }
}
